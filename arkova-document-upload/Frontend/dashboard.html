    <script>

        async function loadStatus() {

            const response = await fetch('/status', {

                headers: { 'Authorization': localStorage.getItem('sessionToken') }

            });

            const statuses = await response.json();

            const tableBody = document.getElementById('statusTable');

            tableBody.innerHTML = statuses.map(s => `

                <tr>

                    <td>${s.docType}</td>

                    <td>${s.customerId}</td>

                    <td>${s.hash}</td>

                    <td>${s.inscriptionId}</td>

                    <td>${s.status}</td>

                </tr>

            `).join('');

        }

        loadStatus();

    </script>
